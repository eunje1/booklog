    <script>
        // 가상의 가짜 도서 검색 API 데이터베이스 (외부 API 응답 지연 및 CORS 제한 방지용)
        const mockBooksDatabase = [
            { id: "1", isbn: "9788937460753", title: "인간 실격", author: "다자이 오사마", publisher: "민음사", cover_url: "https://images.unsplash.com/photo-1544947950-fa07a98d237f?w=150", page_count: 192 },
            { id: "2", isbn: "9788936434267", title: "소년이 온다", author: "한강", publisher: "창비", cover_url: "https://images.unsplash.com/photo-1512820790803-83ca734da794?w=150", page_count: 216 },
            { id: "3", isbn: "9788932917245", title: "사피엔스", author: "유발 하라리", publisher: "김영사", cover_url: "https://images.unsplash.com/photo-1589829545856-d10d557cf95f?w=150", page_count: 636 },
            { id: "4", isbn: "9788954655971", title: "여행의 이유", author: "김영하", publisher: "문학동네", cover_url: "https://images.unsplash.com/photo-1506880018603-83d5b814b5a6?w=150", page_count: 432 }
        ];

        // 상태 관리 변수
        let currentUser = JSON.parse(localStorage.getItem('currentUser')) || null;
        let myLibrary = JSON.parse(localStorage.getItem('myLibrary')) || [];
        let currentSelectedBookId = null;
        let currentFilter = 'all';

        // 뷰 전환 함수
        function switchView(viewId) {
            ['loginView', 'signupView', 'libraryView', 'searchView', 'detailView'].forEach(id => {
                document.getElementById(id).classList.add('hidden-view');
            });
            document.getElementById(viewId).classList.remove('hidden-view');

            if (currentUser) {
                document.getElementById('mainNav').classList.remove('hidden-view');
                document.getElementById('userNickname').innerText = `👋 ${currentUser.nickname}님`;
            } else {
                document.getElementById('mainNav').classList.add('hidden-view');
            }

            if(viewId === 'libraryView') renderLibrary();
        }

        // 회원가입 로직
        function handleSignup() {
            const email = document.getElementById('signupEmail').value;
            const nickname = document.getElementById('signupNickname').value;
            const password = document.getElementById('signupPassword').value;
            const passwordCheck = document.getElementById('signupPasswordCheck').value;
            const errorDiv = document.getElementById('signupError');

            if(!email || !nickname || !password) {
                errorDiv.innerText = "모든 필드를 입력해주세요.";
                errorDiv.classList.remove('hidden-view');
                return;
            }
            if(password !== passwordCheck) {
                errorDiv.innerText = "비밀번호가 일치하지 않습니다.";
                errorDiv.classList.remove('hidden-view');
                return;
            }

            const users = JSON.parse(localStorage.getItem('users')) || [];
            if(users.some(u => u.email === email)) {
                errorDiv.innerText = "이미 존재하는 이메일입니다.";
                errorDiv.classList.remove('hidden-view');
                return;
            }

            users.push({ email, nickname, password });
            localStorage.setItem('users', JSON.stringify(users));
            alert('회원가입 완료! 로그인 해주세요.');
            switchView('loginView');
        }

        // 로그인 로직
        function handleLogin() {
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;
            const errorDiv = document.getElementById('loginError');

            const users = JSON.parse(localStorage.getItem('users')) || [];
            const user = users.find(u => u.email === email && u.password === password);

            if(user) {
                currentUser = user;
                localStorage.setItem('currentUser', JSON.stringify(user));
                switchView('libraryView');
            } else {
                errorDiv.innerText = "이메일 또는 비밀번호를 확인하세요.";
                errorDiv.classList.remove('hidden-view');
            }
        }

        // 로그아웃
        function handleLogout() {
            currentUser = null;
            localStorage.removeItem('currentUser');
            switchView('loginView');
        }

        // 가짜 데이터 기반 검색 함수
        function searchBooks() {
            const kw = document.getElementById('searchKeyword').value.toLowerCase();
            const resultsDiv = document.getElementById('searchResults');
            resultsDiv.innerHTML = '';

            const filtered = mockBooksDatabase.filter(b => b.title.toLowerCase().includes(kw) || b.author.toLowerCase().includes(kw));

            if(filtered.length === 0) {
                resultsDiv.innerHTML = '<p class="text-slate-400 text-sm py-4">검색 결과가 없습니다. (힌트: 인간 실격, 소년이 온다, 사피엔스, 여행의 이유)</p>';
                return;
            }

            filtered.forEach(book => {
                const isAdded = myLibrary.some(b => b.id === book.id);
                resultsDiv.innerHTML += `
                    <div class="bg-white p-4 rounded-xl border border-slate-200 flex gap-4 items-center shadow-sm">
                        <img src="${book.cover_url}" class="w-16 h-24 object-cover bg-slate-100 rounded shadow-sm">
                        <div class="flex-1">
                            <h4 class="font-bold text-base">${book.title}</h4>
                            <p class="text-xs text-slate-500">${book.author} | ${book.publisher}</p>
                            <p class="text-xs text-slate-400 mt-1">ISBN: ${book.isbn}</p>
                            <button onclick="addToLibrary('${book.id}')" ${isAdded ? 'disabled' : ''} class="mt-2 text-xs font-bold px-3 py-1.5 rounded-md ${isAdded ? 'bg-slate-100 text-slate-400' : 'bg-indigo-600 text-white hover:bg-indigo-700'}">
                                ${isAdded ? '이미 서재에 있음' : '내 서재에 추가'}
                            </button>
                        </div>
                    </div>
                `;
            });
        }

        // 내 서재에 추가
        function addToLibrary(bookId) {
            if(myLibrary.some(b => b.id === bookId)) return;
            const bookSrc = mockBooksDatabase.find(b => b.id === bookId);
            
            const newBook = {
                ...bookSrc,
                status: '읽을 책',
                current_page: 0,
                start_date: '',
                end_date: '',
                review: { rating: '5.0', summary: '', content: '' }
            };

            myLibrary.push(newBook);
            localStorage.setItem('myLibrary', JSON.stringify(myLibrary));
            alert(`'${newBook.title}'이(가) 내 서재에 등록되었습니다.`);
            switchView('libraryView');
        }

        // 내 서재 필터링 변경
        function filterLibrary(status) {
            currentFilter = status;
            ['all', '읽을', '읽는', '완독'].forEach(s => {
                const btn = document.getElementById(`btn-filter-${s}`);
                if(s === (status === 'all' ? 'all' : status.substring(0,2))) {
                    btn.className = "px-3 py-1 rounded-full bg-indigo-600 text-white text-sm";
                } else {
                    btn.className = "px-3 py-1 rounded-full bg-slate-200 text-slate-700 text-sm";
                }
            });
            renderLibrary();
        }

        // 내 서재 렌더링
        function renderLibrary() {
            const grid = document.getElementById('libraryGrid');
            grid.innerHTML = '';

            const filtered = currentFilter === 'all' ? myLibrary : myLibrary.filter(b => b.status === currentFilter);

            if(filtered.length === 0) {
                grid.innerHTML = `<div class="col-span-full text-center py-12 text-slate-400">서재가 비어있거나 해당하는 책이 없습니다.</div>`;
                return;
            }

            filtered.forEach(book => {
                const progress = Math.round((book.current_page / book.page_count) * 100) || 0;
                grid.innerHTML += `
                    <div onclick="openBookDetail('${book.id}')" class="bg-white p-4 rounded-2xl border border-slate-200 shadow-sm hover:shadow-md transition cursor-pointer flex gap-4">
                        <img src="${book.cover_url}" class="w-20 h-28 object-cover rounded shadow-sm">
                        <div class="flex-1 flex flex-col justify-between">
                            <div>
                                <span class="text-[10px] font-bold px-2 py-0.5 rounded-full ${book.status==='완독'?'bg-emerald-100 text-emerald-700':book.status==='읽는 중'?'bg-amber-100 text-amber-700':'bg-slate-100 text-slate-600'}">${book.status}</span>
                                <h3 class="font-bold text-base mt-1 line-clamp-1">${book.title}</h3>
                                <p class="text-xs text-slate-400">${book.author}</p>
                            </div>
                            <div class="w-full">
                                <div class="flex justify-between text-[11px] text-slate-500 mb-1">
                                    <span>진척도</span><span>${progress}%</span>
                                </div>
                                <div class="w-full bg-slate-100 h-1.5 rounded-full overflow-hidden">
                                    <div class="bg-indigo-600 h-full" style="width: ${progress}%"></div>
                                </div>
                            </div>
                        </div>
                    </div>
                `;
            });
        }

        // 책 상세 조회 화면 진입
        function openBookDetail(bookId) {
            currentSelectedBookId = bookId;
            const book = myLibrary.find(b => b.id === bookId);
            
            // 상단 기본 정보 세팅
            document.getElementById('bookDetailHeader').innerHTML = `
                <img src="${book.cover_url}" class="w-24 h-36 object-cover rounded-lg shadow">
                <div>
                    <div class="flex items-center gap-2 mb-1">
                        <select onchange="updateBookStatus(this.value)" class="border border-indigo-200 text-xs font-bold px-2 py-1 rounded bg-indigo-50 text-indigo-700">
                            <option value="읽을 책" ${book.status==='읽을 책'?'selected':''}>읽을 책</option>
                            <option value="읽는 중" ${book.status==='읽는 중'?'selected':''}>읽는 중</option>
                            <option value="완독" ${book.status==='완독'?'selected':''}>완독</option>
                        </select>
                        <button onclick="deleteBook('${book.id}')" class="text-xs text-rose-500 hover:underline ml-4">책 삭제</button>
                    </div>
                    <h2 class="text-xl font-bold">${book.title}</h2>
                    <p class="text-sm text-slate-500">${book.author} | ${book.publisher}</p>
                    <p class="text-xs text-slate-400 mt-1">ISBN: ${book.isbn} | 총 ${book.page_count}p</p>
                </div>
            `;

            // 진척도 필드 세팅
            document.getElementById('currentPages').value = book.current_page;
            document.getElementById('totalPagesSpan').innerText = book.page_count;
            
            // 날짜 필드 세팅
            document.getElementById('startDateInput').value = book.start_date || '';
            document.getElementById('endDateInput').value = book.end_date || '';

            // 서평 필드 세팅
            document.getElementById('reviewRating').value = book.review?.rating || '5.0';
            document.getElementById('reviewSummary').value = book.review?.summary || '';
            document.getElementById('reviewContent').value = book.review?.content || '';

            refreshProgressUI(book);
            updateDates();
            switchView('detailView');
        }

        // 독서 상태 변경 함수
        function updateBookStatus(newStatus) {
            const book = myLibrary.find(b => b.id === currentSelectedBookId);
            book.status = newStatus;
            if(newStatus === '완독') {
                book.current_page = book.page_count;
                document.getElementById('currentPages').value = book.page_count;
            }
            localStorage.setItem('myLibrary', JSON.stringify(myLibrary));
            refreshProgressUI(book);
        }

        // 페이지 진척도 저장
        function updateProgress() {
            const book = myLibrary.find(b => b.id === currentSelectedBookId);
            let inputPage = parseInt(document.getElementById('currentPages').value) || 0;

            if(inputPage > book.page_count) inputPage = book.page_count;
            if(inputPage < 0) inputPage = 0;

            book.current_page = inputPage;
            document.getElementById('currentPages').value = inputPage;

            if(book.current_page === book.page_count) {
                book.status = '완독';
            } else if(book.current_page > 0) {
                book.status = '읽는 중';
            }

            localStorage.setItem('myLibrary', JSON.stringify(myLibrary));
            refreshProgressUI(book);
            openBookDetail(book.id); // 상단 배지 동기화용 리로드
        }

        // 진척도 그래픽 요소 리프레시
        function refreshProgressUI(book) {
            const progress = Math.round((book.current_page / book.page_count) * 100) || 0;
            document.getElementById('detailProgressBar').style.width = `${progress}%`;
            document.getElementById('progressPercent').innerText = `${progress}% 완료`;
        }

        // 날짜 차이 계산 기능
        function updateDates() {
            const book = myLibrary.find(b => b.id === currentSelectedBookId);
            const start = document.getElementById('startDateInput').value;
            const end = document.getElementById('endDateInput').value;

            book.start_date = start;
            book.end_date = end;
            localStorage.setItem('myLibrary', JSON.stringify(myLibrary));

            if(start && end) {
                const sDate = new Date(start);
                const eDate = new Date(end);
                const diffTime = eDate - sDate;
                const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)) + 1;
                
                if(diffDays > 0) {
                    document.getElementById('durationDays').innerText = `소요 일수: ${diffDays}일`;
                } else {
                    document.getElementById('durationDays').innerText = `완독일 오류 (시작일 이후 선택)`;
                }
            } else {
                document.getElementById('durationDays').innerText = `소요 일수: -일`;
            }
        }

        // 서평 데이터 저장
        function saveReview() {
            const book = myLibrary.find(b => b.id === currentSelectedBookId);
            book.review = {
                rating: document.getElementById('reviewRating').value,
                summary: document.getElementById('reviewSummary').value,
                content: document.getElementById('reviewContent').value
            };

            localStorage.setItem('myLibrary', JSON.stringify(myLibrary));
            alert('서평이 영구 저장되었습니다!');
        }

        function deleteBook(bookId) {
            if(confirm('정말 삭제하시겠습니까? 관련 데이터가 모두 소멸합니다.')) {
                myLibrary = myLibrary.filter(b => b.id !== bookId);
                localStorage.setItem('myLibrary', JSON.stringify(myLibrary));
                switchView('libraryView');
            }
        }
